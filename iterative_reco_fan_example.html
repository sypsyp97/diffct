2D Fan Beam Iterative Reconstruction

This example demonstrates how to use the differentiable FanProjectorFunction in an iterative reconstruction pipeline for 2D fan-beam geometry. The Shepp-Logan phantom is used as the test object, and a simple gradient-based optimization is performed to reconstruct the image from simulated projections.

Key steps: - Generate a 2D Shepp-Logan phantom. - Simulate fan-beam projections using the differentiable projector. - Define a PyTorch-based iterative reconstruction model and pipeline. - Optimize the reconstruction to minimize the difference between simulated and measured sinograms. - Visualize the loss curve, original phantom, and reconstructed image.

  2import torch
  3import numpy as np
  4import matplotlib.pyplot as plt
  5import torch.nn as nn
  6import torch.optim as optim
  7from diffct.differentiable import FanProjectorFunction

 10def shepp_logan_2d(Nx, Ny):
 11    Nx = int(Nx)
 12    Ny = int(Ny)
 13    phantom = np.zeros((Nx, Ny), dtype=np.float32)
 14    ellipses = [
 15        (0.0, 0.0, 0.69, 0.92, 0, 1.0),
 16        (0.0, -0.0184, 0.6624, 0.8740, 0, -0.8),
 17        (0.22, 0.0, 0.11, 0.31, -18.0, -0.8),
 18        (-0.22, 0.0, 0.16, 0.41, 18.0, -0.8),
 19        (0.0, 0.35, 0.21, 0.25, 0, 0.7),
 20    ]
 21    cx = (Nx - 1)*0.5
 22    cy = (Ny - 1)*0.5
 23    for ix in range(Nx):
 24        for iy in range(Ny):
 25            xnorm = (ix - cx)/(Nx/2)
 26            ynorm = (iy - cy)/(Ny/2)
 27            val = 0.0
 28            for (x0, y0, a, b, angdeg, ampl) in ellipses:
 29                th = np.deg2rad(angdeg)
 30                xprime = (xnorm - x0)*np.cos(th) + (ynorm - y0)*np.sin(th)
 31                yprime = -(xnorm - x0)*np.sin(th) + (ynorm - y0)*np.cos(th)
 32                if xprime*xprime/(a*a) + yprime*yprime/(b*b) <= 1.0:
 33                    val += ampl
 34            phantom[ix, iy] = val
 35    phantom = np.clip(phantom, 0.0, 1.0)
 36    return phantom

 38class IterativeRecoModel(nn.Module):
 39    def __init__(self, volume_shape, angles, 
 40                 num_detectors, detector_spacing, 
 41                 source_distance, isocenter_distance):
 42        
 43        super().__init__()
 44        self.reco = nn.Parameter(torch.zeros(volume_shape))
 45        self.angles = angles
 46        self.num_detectors = num_detectors
 47        self.detector_spacing = detector_spacing
 48        self.source_distance = source_distance
 49        self.isocenter_distance = isocenter_distance
 50
 51    def forward(self, x):
 52        updated_reco = x + self.reco
 53        current_sino = FanProjectorFunction.apply(updated_reco, self.angles, self.num_detectors, self.detector_spacing, self.source_distance, self.isocenter_distance)
 54        return current_sino, updated_reco
 55
 56class Pipeline:
 57    def __init__(self, lr, volume_shape, angles, 
 58                 num_detectors, detector_spacing, 
 59                 source_distance, isocenter_distance, 
 60                 device, epoches=1000):
 61        
 62        self.epoches = epoches
 63        self.model = IterativeRecoModel(volume_shape, angles,
 64                                        num_detectors, detector_spacing, 
 65                                        source_distance, isocenter_distance).to(device)
 66        
 67        self.optimizer = optim.AdamW(list(self.model.parameters()), lr=lr)
 68        self.loss = nn.MSELoss()
 69
 70    def train(self, input, label):
 71        loss_values = []
 72        for epoch in range(self.epoches):
 73            self.optimizer.zero_grad()
 74            predictions, current_reco = self.model(input)
 75            loss_value = self.loss(predictions, label)
 76            loss_value.backward()
 77            self.optimizer.step()
 78            loss_values.append(loss_value.item())
 79
 80            if epoch % 10 == 0:
 81                print(f"Epoch {epoch}, Loss: {loss_value.item()}")
 82                
 83        return loss_values, self.model
 84
 85def main():
 86    Nx, Ny = 128, 128
 87    phantom_cpu = shepp_logan_2d(Nx, Ny)
 88
 89    num_views = 360
 90    angles_np = np.linspace(0, 2 * math.pi, num_views, endpoint=False).astype(np.float32)
 91
 92    num_detectors = 256
 93    detector_spacing = 0.75
 94    source_distance = 600.0
 95    isocenter_distance = 400.0
 96
 97    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
 98    phantom_torch = torch.tensor(phantom_cpu, device=device)
 99    angles_torch = torch.tensor(angles_np, device=device)
100
101    # Generate the "real" sinogram
102    real_sinogram = FanProjectorFunction.apply(phantom_torch, angles_torch,
103                                               num_detectors, detector_spacing,
104                                               source_distance, isocenter_distance)
105
106    pipeline_instance = Pipeline(lr=1e-1,
107                                 volume_shape=(Ny,Nx),
108                                 angles=angles_torch,
109                                 num_detectors=num_detectors,
110                                 detector_spacing=detector_spacing,
111                                 source_distance=source_distance,
112                                 isocenter_distance=isocenter_distance,
113                                 device=device, epoches=1000)
114
115    ini_guess = torch.zeros_like(phantom_torch)
116
117    loss_values, trained_model = pipeline_instance.train(ini_guess, real_sinogram)
118
119    reco = trained_model(ini_guess)[1].squeeze().cpu().detach().numpy()
120
121    plt.figure()
122    plt.plot(loss_values)
123    plt.title("Loss Curve")
124    plt.xlabel("Epoch")
125    plt.ylabel("Loss")
126    plt.show()
127
128    plt.figure(figsize=(12, 6))
129    plt.subplot(1, 2, 1)
130    plt.imshow(phantom_cpu, cmap="gray")
131    plt.title("Original Phantom")
132    plt.axis("off")
133
134    plt.subplot(1, 2, 2)
135    plt.imshow(reco, cmap="gray")
136    plt.title("Reconstructed")
137    plt.axis("off")
138    plt.show()

141    main()
